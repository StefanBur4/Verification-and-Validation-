# ===== 0. Fehlerfälle beim Start =====
log                         # fehlender Username -> Fehler
logout                      # kein User eingeloggt -> Fehler
unknown_command             # sollte "Unknown command" ausgeben

# ===== 1. Admin loggt sich ein und legt Bücher an =====
log admin
add                         # zu wenige Argumente -> Usage-Hinweis
add abc BADISBN Author 2020 # ISBN "abc" -> NumberFormatException in add
add 1111 Dune Herbert 1965
add 2222 CleanCode Martin 2008
add 3333 Patterns Gamma 1994
list                        # alle Bücher anzeigen (alle available)

# ===== 2. search testen (Hilfe, Fehler, Treffer, kein Treffer) =====
search                      # ohne Filter -> Usage / Hilfe
search -d notANumber        # ungültiges Jahr -> Fehler
search -a Herbert           # Suche nach Author Herbert -> sollte Dune finden
search -t CleanCode -d 2008 # kombinierte Suche -> CleanCode
search -t UnknownTitle      # kein Ergebnis

# ===== 3. check ohne ausgeliehene Bücher =====
check                       # keine ausgeliehenen Bücher -> "No borrowed books ..."

# ===== 4. Admin leiht Buch 1 aus, verlängert, gibt zurück =====
borrow 1                    # admin leiht Buch 1
check                       # admin sieht geliehenes Buch
extend 1                    # einmal verlängern -> ok
check -b                    # nur exceeded Bücher -> sollte Buch 1 zeigen
extend 1                    # zweites Extend -> Fehler (already extended)
return 1                    # zurückgeben
return 1                    # nochmal zurückgeben -> Fehler (nicht ausgeliehen)

# ===== 5. Admin loggt aus, Alice versucht Aktionen ohne Rechte =====
logout
borrow 2                    # kein User eingeloggt -> Fehler
log alice
remove 2                    # nicht admin -> Fehler
borrow 2                    # Alice leiht Buch 2
check                       # Alice sieht nur ihr eigenes geliehenes Buch
extend 2                    # Alice verlängert Buch 2
logout

# ===== 6. Bob versucht Buch von Alice zurückzugeben (falscher Borrower) =====
log bob
return 2                    # Buch 2 gehört Alice -> "borrowed by another user"
logout

# ===== 7. Admin räumt auf und entfernt Bücher =====
log admin
check                       # admin sieht ausgeliehenes Buch 2 (Alice)
remove 99                   # ID existiert nicht -> "No book found"
remove 3                    # Buch 3 wird gelöscht
list                        # zeigt nur noch Buch 1 (frei) und Buch 2 (geliehen)
logout

# ===== 8. Nochmal unbekannter Befehl zum Abschluss =====
foo bar                     # Unknown command
